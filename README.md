- **أتمتة الاختبارات والنشر**: ستوفر لك شهوراً من العمل

## 3. بناء الفريق المناسب
- **مطور خبير في Chrome Extensions**: هذا غير قابل للتفاوض
- **خبير أمان**: البيانات حساسة والمخاطر عالية
- **مصمم UX/UI**: التجربة البسيطة = نجاح المنتج
- **مطور backend متمكن**: النظام السحابي عمود المشروع

## 4. استراتيجية تسويق ذكية
- **ابدأ مع شبكتك الشخصية**: أصدقاء وزملاء سابقون
- **محتوى تعليمي قيم**: كيفية تحسين السيرة الذاتية، نصائح المقابلات
- **شراكات مع الجامعات**: الخريجون هم جمهورك الأساسي
- **قصص نجاح حقيقية**: لا شيء يسوق أفضل من النتائج الفعلية

- **اجمع البيانات من اليوم الأول**: كل نقرة وكل تفاعل مهم

## 📋 نظرة عامة على المشروع

### 🎯 الهدف الرئيسي
تطوير إضافة Chrome متقدمة تقوم بأتمتة عملية التقديم على الوظائف في مواقع الشركات السعودية الكبرى مباشرة، مع نظام backend سحابي لإدارة البيانات والاشتراكات.

### 💡 المفهوم الأساسي
النظام يقوم بـ:
1. **التسجيل التلقائي** في مواقع الشركات (أرامكو، سابك، البنوك، إلخ)
2. **التعامل مع تحديات التحقق** (إيميل، SMS، موافقة يدوية)
3. **التقديم التلقائي** على جميع الوظائف المناسبة
4. **إدارة الحملات** مع إمكانية الإيقاف والاستئناف
5. **حفظ البيانات بأمان** مع التشفير الكامل

### 🎯 الجمهور المستهدف
- **الخريجون الجدد**: للتقديم على أكبر عدد من الشركات
- **الباحثون عن عمل**: لتوسيع فرص التوظيف
- **المهنيون**: للانتقال لشركات أفضل
- **الموارد البشرية**: لفهم سوق العمل (B2B)

-- Users Table
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    subscription_tier VARCHAR(50) DEFAULT 'free',
    subscription_end_date TIMESTAMP,
    encrypted_profile TEXT,
    encryption_key_hash VARCHAR(255),
    consent_settings JSONB,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW(),
    last_login TIMESTAMP,
    is_active BOOLEAN DEFAULT true
);

-- Companies Table  
CREATE TABLE companies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    name_en VARCHAR(255),
    website VARCHAR(255) UNIQUE NOT NULL,
    careers_page VARCHAR(255),
    sector VARCHAR(100),
    size VARCHAR(50),
    priority VARCHAR(20) DEFAULT 'medium',
    registration_config JSONB,
    application_config JSONB,
    technical_details JSONB,
    dom_selectors JSONB,
    status VARCHAR(50) DEFAULT 'active',
    last_updated TIMESTAMP DEFAULT NOW()
);

-- Campaigns Table
CREATE TABLE campaigns (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'active',
    target_companies JSONB,
    execution_plan JSONB,
    progress JSONB,
    results JSONB,
    created_at TIMESTAMP DEFAULT NOW(),
    started_at TIMESTAMP,
    completed_at TIMESTAMP
);

-- Company Registrations Table
CREATE TABLE company_registrations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    campaign_id UUID REFERENCES campaigns(id) ON DELETE CASCADE,
    company_id UUID REFERENCES companies(id),
    status VARCHAR(50) DEFAULT 'pending',
    encrypted_credentials TEXT,
    verification_method VARCHAR(50),
    verification_completed_at TIMESTAMP,
    registration_attempts INTEGER DEFAULT 0,
    last_attempt_at TIMESTAMP,
    error_log JSONB,
    account_url VARCHAR(500),
    notes TEXT
);

-- Job Applications Table
CREATE TABLE job_applications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    registration_id UUID REFERENCES company_registrations(id),
    job_title VARCHAR(500),
    job_department VARCHAR(255),
    job_location VARCHAR(255),
    application_date TIMESTAMP DEFAULT NOW(),
    status VARCHAR(50) DEFAULT 'pending',
    rejection_reason TEXT,
    application_url VARCHAR(500),
    response_received_at TIMESTAMP
);

-- Audit Log Table
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    action VARCHAR(100) NOT NULL,
    resource_type VARCHAR(50),
    resource_id UUID,
    old_values JSONB,
    new_values JSONB,
    ip_address INET,
    user_agent TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

// background/campaign-manager.js
class CampaignManager {
  constructor() {
    this.activeCampaigns = new Map();
    this.executionQueue = [];
    this.workers = [];
    this.maxConcurrentWorkers = 3;
  }

  async createCampaign(userProfile, selectedCompanies, preferences) {
    const campaign = {
      id: this.generateCampaignId(),
      user: userProfile,
      companies: selectedCompanies,
      preferences: preferences,
      status: 'created',
      progress: {
        total: selectedCompanies.length,
        completed: 0,
        inProgress: 0,
        pending: selectedCompanies.length,
        failed: 0
      },
      results: {
        registrations: [],
        applications: [],
        errors: []
      },
      createdAt: Date.now()
    };

    // حفظ الحملة في قاعدة البيانات
    await this.saveCampaign(campaign);
    
    // إرسال إلى الخادم السحابي
    await this.syncCampaignToCloud(campaign);
    
    return campaign;
  }

  async startCampaign(campaignId) {
    const campaign = await this.loadCampaign(campaignId);
    campaign.status = 'running';
    campaign.startedAt = Date.now();

    // إنشاء workers للمعالجة المتوازية
    for (let i = 0; i < this.maxConcurrentWorkers; i++) {
      const worker = new CompanyWorker(campaign, i);
      this.workers.push(worker);
      worker.start();
    }

    this.activeCampaigns.set(campaignId, campaign);
    
    // تحديث واجهة المستخدم
    this.notifyUI('campaign_started', { campaignId, campaign });
  }

  async pauseCampaign(campaignId) {
    const campaign = this.activeCampaigns.get(campaignId);
    if (campaign) {
      campaign.status = 'paused';
      campaign.pausedAt = Date.now();
      
      // إيقاف جميع العمال
      this.workers.forEach(worker => worker.pause());
      
      // حفظ الحالة الحالية
      await this.saveCampaignState(campaign);
    }
  }

  async resumeCampaign(campaignId) {
    const campaign = await this.loadCampaign(campaignId);
    campaign.status = 'running';
    campaign.resumedAt = Date.now();

    // استئناف العمال
    this.workers.forEach(worker => worker.resume());
    
    this.activeCampaigns.set(campaignId, campaign);
  }
}

// background/company-worker.js
class CompanyWorker {
  constructor(campaign, workerId) {
    this.campaign = campaign;
    this.workerId = workerId;
    this.isRunning = false;
    this.isPaused = false;
    this.currentCompany = null;
  }

  async start() {
    this.isRunning = true;
    
    while (this.isRunning && this.hasWork()) {
      if (this.isPaused) {
        await this.sleep(1000);
        continue;
      }

      const company = this.getNextCompany();
      if (company) {
        await this.processCompany(company);
      }
      
      // تأخير بين الشركات
      await this.sleep(this.campaign.preferences.delayBetweenCompanies * 1000);
    }
  }

  async processCompany(company) {
    this.currentCompany = company;
    
    try {
      console.log(`[Worker ${this.workerId}] بدء معالجة شركة: ${company.name}`);
      
      // المرحلة 1: التسجيل
      const registrationResult = await this.handleRegistration(company);
      
      if (registrationResult.requiresVerification) {
        // المرحلة 2: التحقق (قد يتطلب تدخل المستخدم)
        const verificationResult = await this.handleVerification(company, registrationResult);
        
        if (!verificationResult.success) {
          // في انتظار تدخل المستخدم
          this.notifyUserVerificationNeeded(company, verificationResult);
          return;
        }
      }
      
      // المرحلة 3: التقديم على الوظائف
      const applicationResults = await this.handleJobApplications(company);
      
      // حفظ النتائج
      await this.saveCompanyResults(company, {
        registration: registrationResult,
        applications: applicationResults
      });
      
      console.log(`[Worker ${this.workerId}] تمت معالجة شركة: ${company.name}`);
      
    } catch (error) {
      console.error(`[Worker ${this.workerId}] خطأ في معالجة ${company.name}:`, error);
      await this.handleCompanyError(company, error);
    }
  }

  async handleRegistration(company) {
    const automation = this.getCompanyAutomation(company);
    const userData = this.campaign.user;
    
    // محاولة التسجيل
    const result = await automation.registerAccount(userData);
    
    if (result.success) {
      // حفظ بيانات الحساب مشفرة
      await this.saveAccountCredentials(company, result.credentials);
      
      return {
        success: true,
        requiresVerification: result.requiresVerification,
        verificationMethod: result.verificationMethod,
        accountUrl: result.accountUrl
      };
    } else {
      throw new Error(`فشل التسجيل: ${result.error}`);
    }
  }

  async handleVerification(company, registrationResult) {
    const verificationMethod = registrationResult.verificationMethod;
    
    switch (verificationMethod) {
      case 'auto':
        // تحقق تلقائي - انتظار قصير
        await this.sleep(5000);
        return { success: true, method: 'auto' };
        
      case 'email_link':
        // محاولة التحقق التلقائي من الإيميل
        const emailResult = await this.attemptEmailVerification(company);
        if (emailResult.success) {
          return emailResult;
        }
        // فشل التحقق التلقائي - يحتاج تدخل المستخدم
        return { success: false, requiresManualVerification: true };
        
      case 'email_code':
      case 'sms':
      case 'manual_approval':
        // يحتاج تدخل المستخدم
        return { success: false, requiresManualVerification: true };
        
      default:
        throw new Error(`طريقة تحقق غير مدعومة: ${verificationMethod}`);
    }
  }

  async attemptEmailVerification(company) {
    // محاولة الوصول لصندوق الإيميل والعثور على رابط التفعيل
    const emailService = new EmailVerificationService();
    const userEmail = this.campaign.user.email;
    
    // انتظار رسالة التفعيل (10 دقائق كحد أقصى)
    for (let attempt = 0; attempt < 20; attempt++) {
      const verificationEmail = await emailService.findVerificationEmail(
        userEmail, 
        company.name
      );
      
      if (verificationEmail) {
        const activationLink = emailService.extractActivationLink(verificationEmail);
        if (activationLink) {
          // فتح رابط التفعيل
          await this.openVerificationLink(activationLink);
          
          // التحقق من نجاح التفعيل
          const isActivated = await this.checkAccountActivation(company);
          if (isActivated) {
            return { success: true, method: 'email_auto' };
          }
        }
      }
      
      // انتظار 30 ثانية قبل المحاولة التالية
      await this.sleep(30000);
    }
    
    return { success: false, reason: 'email_verification_timeout' };
  }

  async handleJobApplications(company) {
    const automation = this.getCompanyAutomation(company);
    const credentials = await this.getAccountCredentials(company);
    
    // تسجيل الدخول
    await automation.login(credentials);
    
    // اكتشاف الوظائف المتاحة
    const availableJobs = await automation.discoverJobs(this.campaign.preferences.jobFilters);
    
    console.log(`وُجد ${availableJobs.length} وظيفة في ${company.name}`);
    
            const applicationResults = [];
    
    for (const job of availableJobs) {
      try {
        // فحص إذا كان تم التقديم مسبقاً
        const alreadyApplied = await automation.checkIfAlreadyApplied(job);
        if (alreadyApplied) {
          applicationResults.push({
            job: job,
            status: 'already_applied',
            timestamp: Date.now()
          });
          continue;
        }
        
        // التقديم على الوظيفة
        const applicationResult = await automation.applyToJob(job, this.campaign.user);
        
        applicationResults.push({
          job: job,
          status: applicationResult.success ? 'success' : 'rejected',
          reason: applicationResult.reason || null,
          timestamp: Date.now()
        });
        
        // تأخير بين التقديمات
        await this.sleep(this.campaign.preferences.delayBetweenApplications * 1000);
        
      } catch (error) {
        console.error(`خطأ في التقديم على الوظيفة ${job.title}:`, error);
        applicationResults.push({
          job: job,
          status: 'error',
          error: error.message,
          timestamp: Date.now()
        });
      }
    }
    
    return applicationResults;
  }

  getCompanyAutomation(company) {
    // إرجاع وحدة الأتمتة المناسبة للشركة
    const moduleName = company.automationModule || 'generic';
    return new (require(`../content/company-modules/${moduleName}.js`))(company);
  }

class EncryptionService {
  constructor() {
    this.algorithm = 'aes-256-gcm';
    this.keyLength = 32;
    this.ivLength = 16;
    this.tagLength = 16;
  }

  async generateUserMasterKey(password, salt) {
    // اشتقاق مفتاح رئيسي من كلمة مرور المستخدم
    const encoder = new TextEncoder();
    const passwordBuffer = encoder.encode(password);
    const saltBuffer = encoder.encode(salt);
    
    const keyMaterial = await crypto.subtle.importKey(
      'raw',
      passwordBuffer,
      { name: 'PBKDF2' },
      false,
      ['deriveKey']
    );
    
    return await crypto.subtle.deriveKey(
      {
        name: 'PBKDF2',
        salt: saltBuffer,
        iterations: 100000,
        hash: 'SHA-256'
      },
      keyMaterial,
      { name: 'AES-GCM', length: 256 },
      true,
      ['encrypt', 'decrypt']
    );
  }

  async encryptData(data, masterKey) {
    const encoder = new TextEncoder();
    const dataBuffer = encoder.encode(JSON.stringify(data));
    
    const iv = crypto.getRandomValues(new Uint8Array(this.ivLength));
    
    const encrypted = await crypto.subtle.encrypt(
      {
        name: 'AES-GCM',
        iv: iv
      },
      masterKey,
      dataBuffer
    );
    
    // دمج IV مع البيانات المشفرة
    const combined = new Uint8Array(iv.length + encrypted.byteLength);
    combined.set(iv);
    combined.set(new Uint8Array(encrypted), iv.length);
    
    return btoa(String.fromCharCode(...combined));
  }

  async decryptData(encryptedData, masterKey) {
    const combined = new Uint8Array(
      atob(encryptedData).split('').map(char => char.charCodeAt(0))
    );
    
    const iv = combined.slice(0, this.ivLength);
    const encrypted = combined.slice(this.ivLength);
    
    const decrypted = await crypto.subtle.decrypt(
      {
        name: 'AES-GCM',
        iv: iv
      },
      masterKey,
      encrypted
    );
    
    const decoder = new TextDecoder();
    return JSON.parse(decoder.decode(decrypted));
  }

  async encryptCredentials(credentials, campaignKey) {
    return await this.encryptData(credentials, campaignKey);
  }

  async decryptCredentials(encryptedCredentials, campaignKey) {
    return await this.decryptData(encryptedCredentials, campaignKey);
  }

  generateSecurePassword(length = 16) {
    const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*';
    const array = new Uint8Array(length);
    crypto.getRandomValues(array);
    
    return Array.from(array, byte => chars[byte % chars.length]).join('');
  }
}

    } catch (error) {

    }
  }

        await this.updateCompanySelectors(companyId, selectors);
      }
      
      console.log('تم تحديث DOM selectors بنجاح');
      return true;
    } catch (error) {
      console.error('فشل في تحديث DOM selectors:', error);
      return false;
    }
  }

  isNewerVersion(remoteVersion, localVersion) {
    const remote = remoteVersion.split('.').map(Number);
    const local = localVersion.split('.').map(Number);
    
    for (let i = 0; i < Math.max(remote.length, local.length); i++) {
      const r = remote[i] || 0;
      const l = local[i] || 0;
      
      if (r > l) return true;
      if (r < l) return false;
    }
    
    return false;
  }
}
